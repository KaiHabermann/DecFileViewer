% Slide 1: Problem Statement
\begin{frame}
\frametitle{The Problem: Finding Files in GitLab}

\begin{columns}[T]
\begin{column}{0.6\textwidth}
\begin{block}{GitLab Search Limitation}
Only filenames are searchable
\end{block}

\vspace{0.3cm}
\begin{itemize}
    \item DecFiles contain complex physics information
    \item Information not always reflected in filenames:
    \begin{itemize}
        \item Decay chains (B0 $\to$ K$^+$ K$^-$)
        \item Event types and descriptors
        \item Physics Working Groups
    \end{itemize}
\end{itemize}
\end{column}

\begin{column}{0.35\textwidth}
\begin{block}{Example}
Looking for $B^0 \to J/\psi K_S^0 \pi^+ \pi^-$ is not straightforward in GitLab
\end{block}

\vspace{0.3cm}
\begin{block}{Solution}
A searchable interface that indexes file contents, not just filenames
\end{block}
\end{column}
\end{columns}

\end{frame}
